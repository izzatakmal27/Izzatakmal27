    <script>
        let myChart = null;

        // Mathematical Default Parameters
        let L = 1800;    // carrying capacity (max mass in g)
        let k = 0.35;    // growth speed coefficient
        let x0 = 12;     // inflection week

        // Generated Weeks array for pineapple post-forcing (week 4 to 22)
        const startWeek = 4;
        const endWeek = 22;
        const step = 0.5;
        let weeksData = [];

        // Logistic function f(x)
        function f(x) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        // First Derivative f'(x)
        function f_prime(x) {
            const expTerm = Math.exp(-k * (x - x0));
            const numerator = L * k * expTerm;
            const denominator = Math.pow(1 + expTerm, 2);
            return numerator / denominator;
        }

        // Generate biological description based on derivative value & week
        function getLaymanInterpretation(week, mass, rate) {
            const peakWeek = parseFloat(x0);
            
            if (week < peakWeek - 4) {
                return `<strong>Lag/Early Growth Phase (Week ${week}):</strong> At this stage, the derivative is relatively low (${rate.toFixed(1)} g/wk). Cells are mostly dividing internally within the fruit core. The pineapple looks tiny. Focus agronomic actions on keeping nitrogen high and managing pest control (e.g., ants/mealybugs) before fast filling begins.`;
            } else if (week >= peakWeek - 4 && week < peakWeek - 0.5) {
                return `<strong>Accelerating Phase (Week ${week}):</strong> The derivative is spiking up rapidly! The pineapple is absorbing nitrogen and potassium greedily. Cells are expanding massively. Your daily watering schedules are highly crucial now to support rapid weight gain.`;
            } else if (week >= peakWeek - 0.5 && week <= peakWeek + 0.5) {
                return `<strong>Inflection / Maximum Growth Stage (Week ${week}):</strong> This is the golden interval where the derivative \(f'(x)\) peaks at ${rate.toFixed(1)} g/wk! The plant is performing at its absolute peak biological metabolic rate. Soil moisture must be perfectly stable; drought now causes physical skin cracking on MD2 fruits.`;
            } else if (week > peakWeek + 0.5 && week <= peakWeek + 4) {
                return `<strong>Decelerating Phase (Week ${week}):</strong> The rate of change has begun to decrease (${rate.toFixed(1)} g/wk), even though the pineapple is still getting heavier. Cellular expansion is winding down. The starch is beginning to convert into simple sugars. Keep potassium high to boost the fruit's ultimate BRIX sweet index.`;
            } else {
                return `<strong>Maturity & Ripening Stage (Week ${week}):</strong> The derivative is near zero (${rate.toFixed(1)} g/wk), meaning the growth has flattened. The crop is physically mature. Golden-yellow skin transformation is beginning. Prepare harvesting labor, check export logistics, and reduce watering to prevent fruit core rot.`;
            }
        }

        function getStageLabel(week) {
            const peakWeek = parseFloat(x0);
            if (week < peakWeek - 4) return "Lag Phase";
            if (week < peakWeek - 0.5) return "Accelerating Growth";
            if (week <= peakWeek + 0.5) return "Peak Inflection";
            if (week <= peakWeek + 4) return "Deceleration";
            return "Harvest Maturity";
        }

        function generateData() {
            weeksData = [];
            let massArray = [];
            let rateArray = [];

            for (let w = startWeek; w <= endWeek; w += step) {
                weeksData.push(w.toFixed(1));
                massArray.push(f(w).toFixed(1));
                rateArray.push(f_prime(w).toFixed(1));
            }
            return { massArray, rateArray };
        }

        function updateProbeDisplay(weekVal) {
            const w = parseFloat(weekVal);
            const mass = f(w);
            const rate = f_prime(w);

            document.getElementById('probeWeek').innerHTML = `Week ${w.toFixed(1)}`;
            document.getElementById('probeMass').innerHTML = `${mass.toFixed(1)} g`;
            document.getElementById('probeRate').innerHTML = `${rate.toFixed(1)} g/wk`;
            document.getElementById('probeStage').innerHTML = getStageLabel(w);
            document.getElementById('probeExplanation').innerHTML = getLaymanInterpretation(w, mass, rate);

            // Animate card highlight visually
            const panel = document.getElementById('interactivePanel');
            panel.classList.add('scale-[1.01]');
            setTimeout(() => {
                panel.classList.remove('scale-[1.01]');
            }, 150);
        }

        function initChart() {
            const { massArray, rateArray } = generateData();
            const ctx = document.getElementById('growthChart').getContext('2d');

            myChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: weeksData,
                    datasets: [
                        {
                            label: 'Fruit Mass f(x) (grams)',
                            data: massArray,
                            borderColor: '#16a34a',
                            backgroundColor: 'rgba(22, 163, 74, 0.05)',
                            borderWidth: 3,
                            fill: true,
                            tension: 0.3,
                            yAxisID: 'yMass',
                            pointRadius: 3,
                            pointHoverRadius: 8,
                            pointHoverBackgroundColor: '#14532d',
                            pointHoverBorderColor: '#ffffff'
                        },
                        {
                            label: "Growth Rate f'(x) (grams/week)",
                            data: rateArray,
                            borderColor: '#f59e0b',
                            backgroundColor: 'transparent',
                            borderWidth: 3.5,
                            borderDash: [5, 5],
                            tension: 0.3,
                            yAxisID: 'yRate',
                            pointRadius: 3,
                            pointHoverRadius: 8,
                            pointHoverBackgroundColor: '#d97706',
                            pointHoverBorderColor: '#ffffff'
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    interaction: {
                        mode: 'index',
                        intersect: false,
                    },
                    plugins: {
                        legend: {
                            display: false // Using custom top legends instead
                        },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    let label = context.dataset.label || '';
                                    if (label) {
                                        label += ': ';
                                    }
                                    if (context.parsed.y !== null) {
                                        label += context.parsed.y;
                                    }
                                    return label;
                                }
                            }
                        }
                    },
                    onClick: (e) => {
                        const points = myChart.getElementsAtEventForMode(e, 'index', { intersect: false }, true);
                        if (points.length > 0) {
                            const index = points[0].index;
                            const weekVal = weeksData[index];
                            updateProbeDisplay(weekVal);
                        }
                    },
                    scales: {
                        x: {
                            title: {
                                display: true,
                                text: 'Time After Flower Induction (Weeks)',
                                color: '#475569',
                                font: {
                                    weight: 'bold',
                                    size: 11
                                }
                            },
                            grid: {
                                color: '#f1f5f9'
                            }
                        },
                        yMass: {
                            type: 'linear',
                            position: 'left',
                            title: {
                                display: true,
                                text: 'Pineapple Fruit Mass (g)',
                                color: '#16a34a',
                                font: {
                                    weight: 'bold',
                                    size: 11
                                }
                            },
                            grid: {
                                color: '#f1f5f9'
                            }
                        },
                        yRate: {
                            type: 'linear',
                            position: 'right',
                            title: {
                                display: true,
                                text: 'Growth Velocity df/dx (g/week)',
                                color: '#d97706',
                                font: {
                                    weight: 'bold',
                                    size: 11
                                }
                            },
                            grid: {
                                drawOnChartArea: false // prevent duplicate grid lines
                            }
                        }
                    }
                }
            });
        }

        // Listeners for Sliders
        const sliderL = document.getElementById('sliderL');
        const sliderK = document.getElementById('sliderK');
        const sliderX0 = document.getElementById('sliderX0');

        const labelL = document.getElementById('labelL');
        const labelK = document.getElementById('labelK');
        const labelX0 = document.getElementById('labelX0');

        function updateChartData() {
            L = parseFloat(sliderL.value);
            k = parseFloat(sliderK.value);
            x0 = parseFloat(sliderX0.value);

            labelL.innerText = `${L} g`;
            labelK.innerText = `${k}`;
            labelX0.innerText = `${x0} Weeks`;

            const { massArray, rateArray } = generateData();
            myChart.data.datasets[0].data = massArray;
            myChart.data.datasets[1].data = rateArray;
            myChart.update();

            // Auto-update probe default based on current x0 location (dynamic midpoint)
            updateProbeDisplay(x0);
        }

        sliderL.addEventListener('input', updateChartData);
        sliderK.addEventListener('input', updateChartData);
        sliderX0.addEventListener('input', updateChartData);

        // Run on Page Load
        window.onload = function() {
            initChart();
            // Default select midpoint to show the absolute peak derivative initially
            updateProbeDisplay(12.0);
        };
    </script>
